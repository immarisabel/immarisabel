    <title>RSS Reader</title>
</head>
<body>
    <div style="float: left; width: 30%;">
        <h2>Blogs</h2>
        <ul>
            <?php
            $json = file_get_contents('https://raw.githubusercontent.com/yourusername/yourrepository/main/blogs.json');
            $blogs = json_decode($json, true);
            foreach ($blogs as $blog) {
                echo '<li><a href="?link=' . urlencode($blog['link']) . '">' . $blog['title'] . '</a></li>';
            }
            ?>
        </ul>
    </div>
    <div style="float: right; width: 70%;">
        <h2>Blog Post</h2>
        <?php
        if (isset($_GET['link'])) {
            $selectedLink = urldecode($_GET['link']);
            echo '<iframe src="' . $selectedLink . '" style="width: 100%; height: 500px;"></iframe>';
        } else {
            echo '<p>Select a blog from the list to read its post.</p>';
        }
        ?>
    </div>
</body>
